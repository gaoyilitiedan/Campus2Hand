## 3. DDL 建表脚本

### 3.1 数据库创建

```sql
CREATE DATABASE IF NOT EXISTS campus2hand
  DEFAULT CHARACTER SET utf8mb4
  DEFAULT COLLATE utf8mb4_unicode_ci;

USE campus2hand;
```

### 3.2 用户表 `users`

```sql
CREATE TABLE `users` (
  `id`            BIGINT        NOT NULL AUTO_INCREMENT  COMMENT '用户ID',
  `student_id`    VARCHAR(12)   NOT NULL                 COMMENT '学号，8-12位数字',
  `password_hash` VARCHAR(128)  NOT NULL                 COMMENT 'bcrypt 密码哈希',
  `nickname`      VARCHAR(30)   NOT NULL                 COMMENT '昵称',
  `avatar_url`    VARCHAR(512)  DEFAULT ''               COMMENT '头像URL',
  `campus_email`  VARCHAR(100)  NOT NULL                 COMMENT '校园邮箱',
  `phone`         VARCHAR(32)   DEFAULT NULL             COMMENT '手机号（AES加密存储，注册后可选绑定）',
  `campus`        VARCHAR(30)   NOT NULL                 COMMENT '所在校区',
  `reputation_score` DECIMAL(2,1) DEFAULT 3.0            COMMENT '信誉分，默认3.0',
  `total_sold`    INT           DEFAULT 0                COMMENT '累计出售数',
  `total_bought`  INT           DEFAULT 0                COMMENT '累计购买数',
  `review_count`  INT           DEFAULT 0                COMMENT '累计评价数',
  `is_verified`   TINYINT(1)    DEFAULT 0                COMMENT '是否已实名认证 0-否 1-是',
  `is_locked`     TINYINT(1)    DEFAULT 0                COMMENT '是否锁定 0-否 1-是',
  `locked_until`  DATETIME      DEFAULT NULL             COMMENT '锁定截止时间',
  `created_at`    DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at`    DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_student_id` (`student_id`),
  UNIQUE KEY `uk_campus_email` (`campus_email`),
  KEY `idx_nickname` (`nickname`),
  KEY `idx_campus` (`campus`),
  KEY `idx_reputation` (`reputation_score`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='用户表';
```

### 3.3 商品表 `products`

```sql
CREATE TABLE `products` (
  `id`          BIGINT        NOT NULL AUTO_INCREMENT  COMMENT '商品ID',
  `seller_id`   BIGINT        NOT NULL                 COMMENT '卖家ID',
  `title`       VARCHAR(60)   NOT NULL                 COMMENT '商品标题',
  `description` VARCHAR(2000) NOT NULL                 COMMENT '商品描述',
  `images`      JSON          NOT NULL                 COMMENT '图片URL数组，最多9张',
  `price`       BIGINT        NOT NULL                 COMMENT '售价（单位：分）',
  `condition`   VARCHAR(20)   NOT NULL                 COMMENT '新旧程度 brand_new/like_new/slightly_used/visibly_used',
  `category`    VARCHAR(20)   NOT NULL                 COMMENT '分类 textbook/digital/lifestyle/sports/clothing/other',
  `campus`      VARCHAR(30)   NOT NULL                 COMMENT '所在校区',
  `status`      VARCHAR(20)   NOT NULL DEFAULT 'pending_review' COMMENT '状态 on_sale/pending_review/sold/delisted',
  `view_count`  INT           DEFAULT 0                COMMENT '浏览量',
  `version`     INT           DEFAULT 0                COMMENT '乐观锁版本号',
  `deleted`     TINYINT(1)    DEFAULT 0                COMMENT '逻辑删除 0-否 1-是',
  `audit_reason` VARCHAR(500) DEFAULT NULL             COMMENT '审核驳回原因',
  `created_at`  DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at`  DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_seller_status` (`seller_id`, `status`),
  KEY `idx_status_category_price` (`status`, `category`, `price`),
  KEY `idx_created_at` (`created_at`),
  KEY `idx_campus` (`campus`),
  FULLTEXT KEY `ft_title_desc` (`title`, `description`) WITH PARSER ngram
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='商品表';
```

### 3.4 订单表 `orders`

```sql
CREATE TABLE `orders` (
  `id`           BIGINT       NOT NULL AUTO_INCREMENT  COMMENT '订单ID',
  `order_no`     VARCHAR(32)  NOT NULL                 COMMENT '订单号（全局唯一）',
  `buyer_id`     BIGINT       NOT NULL                 COMMENT '买家ID',
  `seller_id`    BIGINT       NOT NULL                 COMMENT '卖家ID',
  `product_id`   BIGINT       NOT NULL                 COMMENT '商品ID',
  `amount`       BIGINT       NOT NULL                 COMMENT '订单金额（单位：分）',
  `status`       VARCHAR(20)  NOT NULL DEFAULT 'pending_payment' COMMENT '状态 pending_payment/paid_escrow/pending_shipment/completed/refunding/refunded/cancelled/settlement_failed',
  `refund_reason` VARCHAR(20)  DEFAULT NULL            COMMENT '退款原因',
  `refund_desc`  VARCHAR(500) DEFAULT NULL             COMMENT '退款说明',
  `refund_images` JSON        DEFAULT NULL             COMMENT '退款举证图片',
  `expire_at`    DATETIME     DEFAULT NULL             COMMENT '支付过期时间',
  `paid_at`      DATETIME     DEFAULT NULL             COMMENT '支付时间',
  `completed_at` DATETIME     DEFAULT NULL             COMMENT '完成时间',
  `version`      INT          DEFAULT 0                COMMENT '乐观锁版本号',
  `created_at`   DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at`   DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_order_no` (`order_no`),
  KEY `idx_buyer_status` (`buyer_id`, `status`),
  KEY `idx_seller_status` (`seller_id`, `status`),
  KEY `idx_status_expire` (`status`, `expire_at`),
  KEY `idx_product_id` (`product_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='订单表';
```

### 3.5 支付记录表 `payments`

```sql
CREATE TABLE `payments` (

  `order_id`      BIGINT       NOT NULL                 COMMENT '订单ID',

  `callback_raw`  JSON         DEFAULT NULL             COMMENT '微信回调原始数据（加密存储）',
  `created_at`    DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at`    DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_order_id` (`order_id`),
```

### 3.14 敏感词库表 `sensitive_words`

```sql
CREATE TABLE `sensitive_words` (
  `id`        BIGINT       NOT NULL AUTO_INCREMENT  COMMENT 'ID',
  `word`      VARCHAR(50)  NOT NULL                 COMMENT '敏感词',
  `level`     VARCHAR(10)  NOT NULL DEFAULT 'block' COMMENT '级别 block-禁止 warn-警告',
  `created_at` DATETIME    NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_word` (`word`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='敏感词库表';
```

### 4.2 全文索引使用

```sql
  AND MATCH(title, description) AGAINST('高等数学' IN BOOLEAN MODE)
```

## 7. 迁移与版本管理

### 7.1 Flyway 迁移脚本命名规范

```
src/main/resources/db/migration/
├── V1.0.0__init_schema.sql             # 全量建表
├── V1.0.1__seed_sensitive_words.sql    # 敏感词种子数据
├── V1.0.2__seed_admin_user.sql         # 管理员初始化
└── V1.1.0__add_index_xxx.sql           # 后续版本变更
```

### 7.2 Flyway 配置

```yaml
spring:
  flyway:
    enabled: true
    locations: classpath:db/migration
    baseline-on-migrate: true
    baseline-version: 0
    encoding: UTF-8
```

### 7.3 数据版本管理原则

| 原则           | 说明                               |
|----------------|------------------------------------|
| 向前兼容       | 新增字段必须有 DEFAULT 值           |
| 不可逆         | DDL 变更不做回滚，仅往前修复         |
| 先从后主       | 读写分离场景，DDL 先在从库执行      |
| 低峰变更       | 大表 DDL 在凌晨业务低峰期执行       |
| pt-online      | 大表加字段使用 pt-online-schema-change |

---

> **文档结束** | 建表脚本可直接在 MySQL 8.0+ 中执行。生产环境建议配合 Flyway 进行版本化迁移管理。
